// import JSON data into D3.js const nodes = data.map(item => ({ id: item.Rootdomain.trim(), Subdomains: item.Subdomains, Emails: item.Emails })); // create edges const links = []; function createLinksFromRelations(domain, relationKey, links, type) { if (domain[relationKey] && Array.isArray(domain[relationKey])) { domain[relationKey].forEach(relatedDomain => { // check if there is already an existing link between 2 nodes const existingLink = links.find(link => link.source === domain.Rootdomain.trim() && link.target === relatedDomain.trim()); if (existingLink) { // append the relation type existingLink.type += `/${type}`; } else { // add new link links.push({ source: domain.Rootdomain.trim(), target: relatedDomain.trim(), type: type }); } }); } } // create edges for each json attribute data.forEach(domain => { createLinksFromRelations(domain, "Whois Ip relation", links, 'ASN'); createLinksFromRelations(domain, "Whois Domain relation", links, 'Whois'); createLinksFromRelations(domain, "Mailserver relation", links, 'Mailserver'); createLinksFromRelations(domain, "Nameserver relation", links, 'Nameserver'); createLinksFromRelations(domain, "Title relation", links, 'Title'); createLinksFromRelations(domain, "Google Adsense relation", links, 'GoogleAdsense'); createLinksFromRelations(domain, "Google Analytics relation", links, 'GoogleAnalytics'); createLinksFromRelations(domain, "Social Media relation", links, 'SocialMedia'); createLinksFromRelations(domain, "Favicon relation", links, 'Favicon'); }); // create an SVG container with zoom functionality const svg = d3.select("svg") .call(d3.zoom().on("zoom", function(event) { svg.attr("transform", event.transform); })) .append("g"); // initialize the force simulation const simulation = d3.forceSimulation(nodes) .force("link", d3.forceLink(links).id(d => d.id).distance(100)) .force("charge", d3.forceManyBody().strength(-100).distanceMax(300)) .force("center", d3.forceCenter(window.innerWidth / 2, (window.innerHeight - 100) / 2)) .on("tick", ticked); const emailCount = d.Emails && d.Emails !== '?' ? parseInt(d.Emails, 10) : 0; const size = subdomainCount * 2 + emailCount; // define node size based on amount of subdomains and emails const radiusScale = d3.scaleLog() .domain([1, 800]) .range([3, 13]); // use the scale for positive sizes, otherwise use the minimum size return size > 0 ? radiusScale(size) : 5; }) .style("fill", "#5b5959") .style("stroke-width", 1) .call(d3.drag() .on("start", dragstarted) .on("drag", dragged) .on("end", dragended)) .on("click", handleNodeClick); // update positions of elements let tickCount = 0; function ticked() { tickCount++; // only update every 3 ticks if (tickCount % 3 !== 0) return; link .attr("x1", d => d.source.x) .attr("y1", d => d.source.y) .attr("x2", d => d.target.x) .attr("y2", d => d.target.y); edgeLabel .attr("x", d => (d.source.x + d.target.x) / 2) .attr("y", d => (d.source.y + d.target.y) / 2); node .attr("cx", d => d.x) .attr("cy", d => d.y); labels .attr("x", d => d.x - 10) .attr("y", d => d.y + 25); } // node drag behavior function dragstarted(event, d) { if (!event.active) simulation.alphaTarget(0.3).restart(); d.fx = d.x; d.fy = d.y; } function dragged(event, d) { d.fx = event.x; d.fy = event.y; } function dragended(event, d) { if (!event.active) simulation.alphaTarget(0); d.fx = d.x; d.fy = d.y; } // handle click on a node function handleNodeClick(event, d) { // get data of clicked node const nodeData = data.find(p => p.Rootdomain.trim() === d.id);
